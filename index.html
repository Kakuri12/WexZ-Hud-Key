-- Table 1: บันทึก Trade ทุกรายการ
create table trades (
  id          bigserial primary key,
  magic       int,
  symbol      text,
  direction   text,
  open_price  float8,
  sl          float8,
  tp          float8,
  lot         float8,
  profit      float8 default 0,
  status      text default 'OPEN',
  ticket      bigint,
  server_time text,
  close_time  text,
  created_at  timestamptz default now()
);
alter table trades enable row level security;
create policy "anon read" on trades for select using (true);
create policy "anon insert" on trades for insert with check (true);
create policy "anon update" on trades for update using (true);

-- Table 2: บันทึก EA Status / Circuit Breaker
create table ea_status (
  id          bigserial primary key,
  magic       int,
  symbol      text,
  status_msg  text,
  balance     float8,
  equity      float8,
  consec_loss int,
  logged_at   text,
  created_at  timestamptz default now()
);
alter table ea_status enable row level security;
create policy "anon all" on ea_status for all using (true) with check (true);
